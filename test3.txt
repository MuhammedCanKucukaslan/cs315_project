/*
* it turns the heading of drone for 
*   degree many degrees, and clockwise if isClockWise is true, 
* counter clockwise otherwise.  
*/
function turnFor(int degree; bool isClockWise) {
    for(int i = 0; i < degree; i++) {
        setHeading( !isClockWise);
    }
}

/* It simulates a robot dance.
*/
function dance() {
    for(int i = 0; i < 100; i++) {
        setVerticalClimb( 1);
        turnFor(180, true);
        sleep(1000);

        setVerticalClimb( 0);
        
        setVerticalClimb( -1);
        turnFor(180, false);
        sleep(1000);        
    }
}
